Integration Code: 
<!-- Include the SDK -->
<script src="https://cdn.pso-gateway.com/sdk/pso-sdk.js"></script>

<script>
  // Initialize the SDK with your merchant ID
  const pso = new PSOPayment({
    merchantId: 'your-merchant-id',
    environment: 'production' // or 'test'
  });

  // Show payment form when user clicks buy
  function buyProduct(amount, productName) {
    pso.showPaymentForm({
      amount: amount, // amount in cents
      currency: 'USD',
      metadata: { productName: productName },
      onSuccess: (result) => {
        // Payment successful - fulfill order
        console.log('Payment successful:', result);
        showNotification('success', 'Payment successful!');
      },
      onError: (error) => {
        // Payment failed - show error
        console.error('Payment failed:', error);
        showNotification('error', error.message);
      }
    });
  }
</script>
